<script>
let notes = JSON.parse(localStorage.getItem("longNotes")) || [];
let active = null;

const editor = document.getElementById("editor");
const list = document.getElementById("list");

function save(){
  localStorage.setItem("longNotes",JSON.stringify(notes));
}

function render(){
  list.innerHTML="";
  notes.forEach(n=>{
    const d=document.createElement("div");
    d.className="note-item";
    d.innerText=n.title || "Tanpa Judul";
    d.onclick=()=>openNote(n.id);
    list.appendChild(d);
  });
}

function newNote(){
  const id=Date.now();
  notes.unshift({
    id,
    title:"Catatan Baru",
    content:""
  });
  openNote(id);
  save();
  render();
}

function openNote(id){
  active=id;
  const n=notes.find(n=>n.id==id);
  editor.value=n.content;
}

editor.addEventListener("input",()=>{
  const n=notes.find(n=>n.id==active);
  if(!n) return;
  n.content=editor.value;
  n.title=editor.value.split("\n")[0];
  save();
  render();
});

render();
</script>
